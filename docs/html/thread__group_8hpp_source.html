42 // Create a new thread in the group.
43 template <typename Function>
44 void create_thread(Function f)
45 {
46 first_ = new item(f, first_);
47 }
48
49 // Create new threads in the group.
50 template <typename Function>
51 void create_threads(Function f, std::size_t num_threads)
52 {
53 for (std::size_t i = 0; i < num_threads; ++i)
54 create_thread(f);
55 }
56
57 // Wait for all threads in the group to exit.
58 void join()
59 {
60 while (first_)
61 {
62 first_->thread_.join();
63 item* tmp = first_;
64 first_ = first_->next_;
65 delete tmp;
66 }
67 }
68
69 // Test whether the group is empty.
70 bool empty() const
71 {
72 return first_ == 0;
73 }
74
75private:

84 }

88 };

90 // The first thread in the group.
