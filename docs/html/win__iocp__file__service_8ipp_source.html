51asio::error_code win_iocp_file_service::open(
52 win_iocp_file_service::implementation_type& impl,
53 const char* path, file_base::flags open_flags,
54 asio::error_code& ec)
55{
56 if (is_open(impl))
57 {
58 ec = asio::error::already_open;
59 ASIO_ERROR_LOCATION(ec);
60 return ec;
61 }
62
63 DWORD access = 0;
64 if ((open_flags & file_base::read_only) != 0)
65 access = GENERIC_READ;
66 else if ((open_flags & file_base::write_only) != 0)
67 access = GENERIC_WRITE;
68 else if ((open_flags & file_base::read_write) != 0)
69 access = GENERIC_READ | GENERIC_WRITE;
70
71 DWORD share = FILE_SHARE_READ | FILE_SHARE_WRITE;
72
73 DWORD disposition = 0;
74 if ((open_flags & file_base::create) != 0)
75 {
76 if ((open_flags & file_base::exclusive) != 0)
77 disposition = CREATE_NEW;
78 else
79 disposition = OPEN_ALWAYS;
80 }
81 else
82 {
83 if ((open_flags & file_base::truncate) != 0)
84 disposition = TRUNCATE_EXISTING;
85 else
86 disposition = OPEN_EXISTING;
87 }
88
89 DWORD flags = FILE_ATTRIBUTE_NORMAL | FILE_FLAG_OVERLAPPED;
90 if (impl.is_stream_)
91 flags |= FILE_FLAG_SEQUENTIAL_SCAN;
92 else
93 flags |= FILE_FLAG_RANDOM_ACCESS;
94 if ((open_flags & file_base::sync_all_on_write) != 0)
95 flags |= FILE_FLAG_WRITE_THROUGH;
96
97 HANDLE handle = ::CreateFileA(path, access, share, 0, disposition, flags, 0);
98 if (handle != INVALID_HANDLE_VALUE)
99 {
100 if (disposition == OPEN_ALWAYS && (open_flags & file_base::truncate) != 0)
101 {
102 if (!::SetEndOfFile(handle))
103 {
104 DWORD last_error = ::GetLastError();
105 ::CloseHandle(handle);
106 ec.assign(last_error, asio::error::get_system_category());
107 ASIO_ERROR_LOCATION(ec);
108 return ec;
109 }
110 }
111
112 handle_service_.assign(impl, handle, ec);
113 if (ec)
114 ::CloseHandle(handle);
115 impl.offset_ = 0;
116 ASIO_ERROR_LOCATION(ec);
117 return ec;
118 }
119 else
120 {
121 DWORD last_error = ::GetLastError();
122 ec.assign(last_error, asio::error::get_system_category());
123 ASIO_ERROR_LOCATION(ec);
124 return ec;
125 }
126}

147asio::error_code win_iocp_file_service::resize(
148 win_iocp_file_service::implementation_type& impl,
149 uint64_t n, asio::error_code& ec)
150{
151 LARGE_INTEGER distance;
152 distance.QuadPart = n;
153 if (::SetFilePointerEx(native_handle(impl), distance, 0, FILE_BEGIN))
154 {
155 BOOL result = ::SetEndOfFile(native_handle(impl));
156 DWORD last_error = ::GetLastError();
157
158 distance.QuadPart = static_cast<LONGLONG>(impl.offset_);
159 if (!::SetFilePointerEx(native_handle(impl), distance, 0, FILE_BEGIN))
160 {
161 result = FALSE;
162 last_error = ::GetLastError();
163 }
164
165 if (result)
166 asio::error::clear(ec);
167 else
168 ec.assign(last_error, asio::error::get_system_category());
169 ASIO_ERROR_LOCATION(ec);
170 return ec;
171 }
172 else
173 {
174 DWORD last_error = ::GetLastError();
175 ec.assign(last_error, asio::error::get_system_category());
176 ASIO_ERROR_LOCATION(ec);
177 return ec;
178 }

217uint64_t win_iocp_file_service::seek(
218 win_iocp_file_service::implementation_type& impl, int64_t offset,
219 file_base::seek_basis whence, asio::error_code& ec)
220{
221 DWORD method;
222 switch (whence)
223 {
224 case file_base::seek_set:
225 method = FILE_BEGIN;
226 break;
227 case file_base::seek_cur:
228 method = FILE_BEGIN;
229 offset = static_cast<int64_t>(impl.offset_) + offset;
230 break;
231 case file_base::seek_end:
232 method = FILE_END;
233 break;
234 default:
235 ec = asio::error::invalid_argument;
236 ASIO_ERROR_LOCATION(ec);
237 return 0;
238 }
239
240 LARGE_INTEGER distance, new_offset;
241 distance.QuadPart = offset;
242 if (::SetFilePointerEx(native_handle(impl), distance, &new_offset, method))
243 {
244 impl.offset_ = new_offset.QuadPart;
245 asio::error::clear(ec);
246 return impl.offset_;
247 }
248 else
249 {
250 DWORD last_error = ::GetLastError();
251 ec.assign(last_error, asio::error::get_system_category());
252 ASIO_ERROR_LOCATION(ec);
253 return 0;
254 }
255}
