61struct win_iocp_io_context::timer_thread_function
62{
63 void operator()()
64 {
65 while (::InterlockedExchangeAdd(&io_context_->shutdown_, 0) == 0)
66 {
67 if (::WaitForSingleObject(io_context_->waitable_timer_.handle,
68 INFINITE) == WAIT_OBJECT_0)
69 {
70 ::InterlockedExchange(&io_context_->dispatch_required_, 1);
71 ::PostQueuedCompletionStatus(io_context_->iocp_.handle,
72 0, wake_for_dispatch, 0);
73 }
74 }
75 }

80win_iocp_io_context::win_iocp_io_context(
81 asio::execution_context& ctx, int concurrency_hint, bool own_thread)
82 : execution_context_service_base<win_iocp_io_context>(ctx),
83 iocp_(),
84 outstanding_work_(0),
85 stopped_(0),
86 stop_event_posted_(0),
87 shutdown_(0),
88 gqcs_timeout_(get_gqcs_timeout()),
89 dispatch_required_(0),
90 concurrency_hint_(concurrency_hint)
91{
92 ASIO_HANDLER_TRACKING_INIT;
93
94 iocp_.handle = ::CreateIoCompletionPort(INVALID_HANDLE_VALUE, 0, 0,
95 static_cast<DWORD>(concurrency_hint >= 0 ? concurrency_hint : DWORD(~0)));
96 if (!iocp_.handle)
97 {
98 DWORD last_error = ::GetLastError();
99 asio::error_code ec(last_error,
100 asio::error::get_system_category());
101 asio::detail::throw_error(ec, "iocp");
102 }
103
104 if (own_thread)
105 {
106 ::InterlockedIncrement(&outstanding_work_);
107 thread_.reset(new asio::detail::thread(thread_function(this)));
108 }
109}

121void win_iocp_io_context::shutdown()
122{
123 ::InterlockedExchange(&shutdown_, 1);
124
125 if (timer_thread_.get())
126 {
127 LARGE_INTEGER timeout;
128 timeout.QuadPart = 1;
129 ::SetWaitableTimer(waitable_timer_.handle, &timeout, 1, 0, 0, FALSE);
130 }
131
132 if (thread_.get())
133 {
134 stop();
135 thread_->join();
136 thread_.reset();
137 ::InterlockedDecrement(&outstanding_work_);
138 }
139
140 while (::InterlockedExchangeAdd(&outstanding_work_, 0) > 0)
141 {
142 op_queue<win_iocp_operation> ops;
143 timer_queues_.get_all_timers(ops);
144 ops.push(completed_ops_);
145 if (!ops.empty())
146 {
147 while (win_iocp_operation* op = ops.front())
148 {
149 ops.pop();
150 ::InterlockedDecrement(&outstanding_work_);
151 op->destroy();
152 }
153 }
154 else
155 {
156 DWORD bytes_transferred = 0;
157 dword_ptr_t completion_key = 0;
158 LPOVERLAPPED overlapped = 0;
159 ::GetQueuedCompletionStatus(iocp_.handle, &bytes_transferred,
160 &completion_key, &overlapped, gqcs_timeout_);
161 if (overlapped)
162 {
163 ::InterlockedDecrement(&outstanding_work_);
164 static_cast<win_iocp_operation*>(overlapped)->destroy();
165 }
166 }
167 }
168
169 if (timer_thread_.get())
170 {
171 timer_thread_->join();
172 timer_thread_.reset();
173 }
174}

192size_t win_iocp_io_context::run(asio::error_code& ec)
193{
194 if (::InterlockedExchangeAdd(&outstanding_work_, 0) == 0)
195 {
196 stop();
197 ec = asio::error_code();
198 return 0;
199 }
200
201 win_iocp_thread_info this_thread;
202 thread_call_stack::context ctx(this, this_thread);
203
204 size_t n = 0;
205 while (do_one(INFINITE, this_thread, ec))
206 if (n != (std::numeric_limits<size_t>::max)())
207 ++n;
208 return n;
209}

303void win_iocp_io_context::post_deferred_completion(win_iocp_operation* op)
304{
305 // Flag the operation as ready.
306 op->ready_ = 1;
307
308 // Enqueue the operation on the I/O completion port.
309 if (!::PostQueuedCompletionStatus(iocp_.handle, 0, 0, op))
310 {
311 // Out of resources. Put on completed queue instead.
312 mutex::scoped_lock lock(dispatch_mutex_);
313 completed_ops_.push(op);
314 ::InterlockedExchange(&dispatch_required_, 1);
315 }
316}

412size_t win_iocp_io_context::do_one(DWORD msec,
413 win_iocp_thread_info& this_thread, asio::error_code& ec)
414{
415 for (;;)
416 {
417 // Try to acquire responsibility for dispatching timers and completed ops.
418 if (::InterlockedCompareExchange(&dispatch_required_, 0, 1) == 1)
419 {
420 mutex::scoped_lock lock(dispatch_mutex_);
421
422 // Dispatch pending timers and operations.
423 op_queue<win_iocp_operation> ops;
424 ops.push(completed_ops_);
425 timer_queues_.get_ready_timers(ops);
426 post_deferred_completions(ops);
427 update_timeout();
428 }
429
430 // Get the next operation from the queue.
431 DWORD bytes_transferred = 0;
432 dword_ptr_t completion_key = 0;
433 LPOVERLAPPED overlapped = 0;
434 ::SetLastError(0);
435 BOOL ok = ::GetQueuedCompletionStatus(iocp_.handle,
436 &bytes_transferred, &completion_key, &overlapped,
437 msec < gqcs_timeout_ ? msec : gqcs_timeout_);
438 DWORD last_error = ::GetLastError();
439
440 if (overlapped)
441 {
442 win_iocp_operation* op = static_cast<win_iocp_operation*>(overlapped);
443 asio::error_code result_ec(last_error,
444 asio::error::get_system_category());
445
446 // We may have been passed the last_error and bytes_transferred in the
447 // OVERLAPPED structure itself.
448 if (completion_key == overlapped_contains_result)
449 {
450 result_ec = asio::error_code(static_cast<int>(op->Offset),
451 *reinterpret_cast<asio::error_category*>(op->Internal));
452 bytes_transferred = op->OffsetHigh;
453 }
454
455 // Otherwise ensure any result has been saved into the OVERLAPPED
456 // structure.
457 else
458 {
459 op->Internal = reinterpret_cast<ulong_ptr_t>(&result_ec.category());
460 op->Offset = result_ec.value();
461 op->OffsetHigh = bytes_transferred;
462 }
463
464 // Dispatch the operation only if ready. The operation may not be ready
465 // if the initiating function (e.g. a call to WSARecv) has not yet
466 // returned. This is because the initiating function still wants access
467 // to the operation's OVERLAPPED structure.
468 if (::InterlockedCompareExchange(&op->ready_, 1, 0) == 1)
469 {
470 // Ensure the count of outstanding work is decremented on block exit.
471 work_finished_on_block_exit on_exit = { this };
472 (void)on_exit;
473
474 op->complete(this, result_ec, bytes_transferred);
475 this_thread.rethrow_pending_exception();
476 ec = asio::error_code();
477 return 1;
478 }
479 }
480 else if (!ok)
481 {
482 if (last_error != WAIT_TIMEOUT)
483 {
484 ec = asio::error_code(last_error,
485 asio::error::get_system_category());
486 return 0;
487 }
488
489 // If we're waiting indefinitely we need to keep going until we get a
490 // real handler.
491 if (msec == INFINITE)
492 continue;
493
494 ec = asio::error_code();
495 return 0;
496 }
497 else if (completion_key == wake_for_dispatch)
498 {
499 // We have been woken up to try to acquire responsibility for dispatching
500 // timers and completed operations.
501 }
502 else
503 {
504 // Indicate that there is no longer an in-flight stop event.
505 ::InterlockedExchange(&stop_event_posted_, 0);
506
507 // The stopped_ flag is always checked to ensure that any leftover
508 // stop events from a previous run invocation are ignored.
509 if (::InterlockedExchangeAdd(&stopped_, 0) != 0)
510 {
511 // Wake up next thread that is blocked on GetQueuedCompletionStatus.
512 if (::InterlockedExchange(&stop_event_posted_, 1) == 0)
513 {
514 if (!::PostQueuedCompletionStatus(iocp_.handle, 0, 0, 0))
515 {
516 last_error = ::GetLastError();
517 ec = asio::error_code(last_error,
518 asio::error::get_system_category());
519 return 0;
520 }
521 }
522
523 ec = asio::error_code();
524 return 0;
525 }
526 }
527 }

530DWORD win_iocp_io_context::get_gqcs_timeout()
531{
532#if !defined(_WIN32_WINNT) || (_WIN32_WINNT < 0x0600)
533 OSVERSIONINFOEX osvi;
534 ZeroMemory(&osvi, sizeof(osvi));
535 osvi.dwOSVersionInfoSize = sizeof(osvi);
536 osvi.dwMajorVersion = 6ul;
537
538 const uint64_t condition_mask = ::VerSetConditionMask(
539 0, VER_MAJORVERSION, VER_GREATER_EQUAL);
540
541 if (!!::VerifyVersionInfo(&osvi, VER_MAJORVERSION, condition_mask))
542 return INFINITE;
543
544 return default_gqcs_timeout;
545#else // !defined(_WIN32_WINNT) || (_WIN32_WINNT < 0x0600)
546 return INFINITE;
547#endif // !defined(_WIN32_WINNT) || (_WIN32_WINNT < 0x0600)
548}

550void win_iocp_io_context::do_add_timer_queue(timer_queue_base& queue)
551{
552 mutex::scoped_lock lock(dispatch_mutex_);
553
554 timer_queues_.insert(&queue);
555
556 if (!waitable_timer_.handle)
557 {
558 waitable_timer_.handle = ::CreateWaitableTimer(0, FALSE, 0);
559 if (waitable_timer_.handle == 0)
560 {
561 DWORD last_error = ::GetLastError();
562 asio::error_code ec(last_error,
563 asio::error::get_system_category());
564 asio::detail::throw_error(ec, "timer");
565 }
566
567 LARGE_INTEGER timeout;
568 timeout.QuadPart = -max_timeout_usec;
569 timeout.QuadPart *= 10;
570 ::SetWaitableTimer(waitable_timer_.handle,
571 &timeout, max_timeout_msec, 0, 0, FALSE);
572 }
573
574 if (!timer_thread_.get())
575 {
576 timer_thread_function thread_function = { this };
577 timer_thread_.reset(new thread(thread_function, 65536));
578 }
579}

588void win_iocp_io_context::update_timeout()
589{
590 if (timer_thread_.get())
591 {
592 // There's no point updating the waitable timer if the new timeout period
593 // exceeds the maximum timeout. In that case, we might as well wait for the
594 // existing period of the timer to expire.
595 long timeout_usec = timer_queues_.wait_duration_usec(max_timeout_usec);
596 if (timeout_usec < max_timeout_usec)
597 {
598 LARGE_INTEGER timeout;
599 timeout.QuadPart = -timeout_usec;
600 timeout.QuadPart *= 10;
601 ::SetWaitableTimer(waitable_timer_.handle,
602 &timeout, max_timeout_msec, 0, 0, FALSE);
603 }
604 }
