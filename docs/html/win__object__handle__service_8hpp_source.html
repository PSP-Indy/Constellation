136 template <typename Handler, typename IoExecutor>
137 void async_wait(implementation_type& impl,
138 Handler& handler, const IoExecutor& io_ex)
139 {
140 // Allocate and construct an operation to wrap the handler.
141 typedef wait_handler<Handler, IoExecutor> op;
142 typename op::ptr p = { asio::detail::addressof(handler),
143 op::ptr::allocate(handler), 0 };
144 p.p = new (p.v) op(handler, io_ex);
145
146 ASIO_HANDLER_CREATION((scheduler_.context(), *p.p, "object_handle",
147 &impl, reinterpret_cast<uintmax_t>(impl.wait_handle_), "async_wait"));
148
149 start_wait_op(impl, p.p);
150 p.v = p.p = 0;
151 }
