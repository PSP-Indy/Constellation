74inline long clamp_thread_pool_size(std::size_t n)
75{
76 if (n > 0x7FFFFFFF)
77 {
78 std::out_of_range ex("thread pool size");
79 asio::detail::throw_exception(ex);
80 }
81 return static_cast<long>(n & 0x7FFFFFFF);

125detail::scheduler& thread_pool::add_scheduler(detail::scheduler* s)
126{
127 detail::scoped_ptr<detail::scheduler> scoped_impl(s);
128 asio::add_service<detail::scheduler>(*this, scoped_impl.get());
129 return *scoped_impl.release();
