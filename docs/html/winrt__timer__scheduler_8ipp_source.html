49void winrt_timer_scheduler::shutdown()
50{
51 asio::detail::mutex::scoped_lock lock(mutex_);
52 shutdown_ = true;
53 stop_thread_ = true;
54 event_.signal(lock);
55 lock.unlock();
56
57 if (thread_)
58 {
59 thread_->join();
60 delete thread_;
61 thread_ = 0;
62 }
63
64 op_queue<operation> ops;
65 timer_queues_.get_all_timers(ops);
66 scheduler_.abandon_operations(ops);
67}

77void winrt_timer_scheduler::run_thread()
78{
79 asio::detail::mutex::scoped_lock lock(mutex_);
80 while (!stop_thread_)
81 {
82 const long max_wait_duration = 5 * 60 * 1000000;
83 long wait_duration = timer_queues_.wait_duration_usec(max_wait_duration);
84 event_.wait_for_usec(lock, wait_duration);
85 event_.clear(lock);
86 op_queue<operation> ops;
87 timer_queues_.get_ready_timers(ops);
88 if (!ops.empty())
89 {
90 lock.unlock();
91 scheduler_.post_deferred_completions(ops);
92 lock.lock();
93 }
94 }
95}
